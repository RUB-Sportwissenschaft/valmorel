JSONBin.io API-Details:
- Daten lesen: GET https://api.jsonbin.io/v3/b/{BIN_ID}/latest
  Response: { record: [...] }
- Daten aktualisieren: PUT https://api.jsonbin.io/v3/b/{BIN_ID}
  Body: JSON-Array aller Einträge
- Header bei beiden: X-Master-Key: {JSONBIN_API_KEY}
- Bin-Strategie: Ein einzelner Bin mit Array aller Antworten.
  Beim Absenden: GET aktuelle Liste → Eintrag pushen → PUT zurückschreiben.
- Antwort-Objekt: { vorname, nachname, gruppe, wahl, timestamp: new Date().toISOString() }

```javascript
function initValidierung() {

  document.getElementById('nachname').addEventListener('input', validateForm);

function validateForm() {
  const vorname = document.getElementById('vorname').value.trim();
  const nachname = document.getElementById('nachname').value.trim();
  const gruppeSelected = document.querySelector('input[name="gruppe"]:checked');
  const wahlSelected = document.querySelector('input[name="wahl"]:checked');
  const valid = vorname.length >= 2 && nachname.length >= 2
    && gruppeSelected !== null && wahlSelected !== null;
  document.getElementById('submit-btn').disabled = !valid;
}
```

```javascript
function initSubmit() {
  document.getElementById('umfrage-form').addEventListener('submit', async (e) => {
    e.preventDefault();

    const vorname = document.getElementById('vorname').value.trim();
    const nachname = document.getElementById('nachname').value.trim();
    const gruppe = document.querySelector('input[name="gruppe"]:checked')?.value;
    const wahl = document.querySelector('input[name="wahl"]:checked')?.value;

    let hasError = false;

    if (vorname.length < 2) {
      document.getElementById('vorname-error').textContent = 'Bitte Vorname eingeben (mindestens 2 Zeichen).';
      hasError = true;
    } else {
      document.getElementById('vorname-error').textContent = '';
    }

    if (nachname.length < 2) {
      document.getElementById('nachname-error').textContent = 'Bitte Nachname eingeben (mindestens 2 Zeichen).';
      hasError = true;
    } else {
      document.getElementById('nachname-error').textContent = '';
    }

    if (!gruppe) {
      document.getElementById('gruppe-error').textContent = 'Bitte eine Gruppe auswählen.';
      hasError = true;
    } else {
      document.getElementById('gruppe-error').textContent = '';
    }

    if (!wahl) {
      document.getElementById('wahl-error').textContent = 'Bitte eine Essensoption auswählen.';
      hasError = true;
    } else {
      document.getElementById('wahl-error').textContent = '';
    }

    if (hasError) return;

    await saveToJsonBin({ vorname, nachname, gruppe, wahl });
  });
}
```

```javascript
async function saveToJsonBin(antwort) {
  const btn = document.getElementById('submit-btn');
  btn.textContent = 'Speichern\u2026';
  btn.disabled = true;
  document.getElementById('network-error').textContent = '';

  const eintrag = {
    vorname: antwort.vorname,
    nachname: antwort.nachname,
    gruppe: antwort.gruppe,
    wahl: antwort.wahl,
    timestamp: new Date().toISOString()
  };

  try {
    // Schritt 1: Aktuelle Liste laden
    const getRes = await fetch(`https://api.jsonbin.io/v3/b/${JSONBIN_BIN_ID}/latest`, {
      headers: { 'X-Master-Key': JSONBIN_API_KEY }
    });
    let liste = [];
    if (getRes.ok) {
      const data = await getRes.json();
      liste = Array.isArray(data.record) ? data.record : [];
    }

    // Schritt 2: Neuen Eintrag anhängen
    liste.push(eintrag);

    // Schritt 3: Zurückschreiben
    const putRes = await fetch(`https://api.jsonbin.io/v3/b/${JSONBIN_BIN_ID}`, {
      method: 'PUT',
      headers: {
        'Content-Type': 'application/json',
        'X-Master-Key': JSONBIN_API_KEY
      },
      body: JSON.stringify(liste)
    });

    if (!putRes.ok) throw new Error(`HTTP ${putRes.status}`);

    // Erfolg
    sessionStorage.setItem('valmorel_voted', '1');
    showConfirmation(antwort);

  } catch (err) {
    btn.textContent = 'Absenden';
    btn.disabled = false;
    document.getElementById('network-error').textContent =
      'Speichern fehlgeschlagen. Bitte Internetverbindung prüfen und erneut versuchen.';
  }
}
```

```javascript
function showConfirmation(antwort) {
  const wahlLabels = {
    'raclette': '\uD83E\uDDC0 Raclette',
    'raclette-vegetarisch': '\uD83E\uDD66 Raclette vegetarisch',
    'vegan': '\uD83C\uDF31 Vegan'
  };

  document.getElementById('umfrage-form').style.display = 'none';
  const conf = document.getElementById('confirmation');
  conf.style.display = 'block';
  document.getElementById('confirmation-text').textContent =
    `Danke, ${antwort.vorname}! Deine Wahl wurde gespeichert.`;
  document.getElementById('confirmation-choice').innerHTML =
    `<p><strong>Gruppe:</strong> ${antwort.gruppe}</p>
     <p><strong>Essenswahl:</strong> ${wahlLabels[antwort.wahl] || antwort.wahl}</p>`;
}
```
